export const FilterEvent = ({ filterEvent, setFilterEvent }) => {
  /*   // get data to sort on category

    );
    console.log(sortCategories);

  const getData = async () => {
    try {
      // Build the filtering URL based on the filterEvent state
      const url = `http://localhost:3000/events${
        filterEvent
          ? `?categoryIds=${filterEvent}&_sort=categorie.name&_order=asc`
          : ''
      }`;

      const response = await fetch(url);

      if (response.ok) {
        const responseData = await response.json();
        filterEvent(responseData);
      } else {
        throw new Error('Error retrieving data');
      }
    } catch (error) {
      console.log(error);
    }
  };

  return (
    <div>
      <button onClick={() => setFilterEvent('Blues')}>Blues</button>
      <button onClick={() => setFilterEvent('Jazz')}>Jazz</button>
      <button onClick={() => setFilterEvent('Pop')}>Pop</button>
      <button onClick={() => setFilterEvent('Rock')}>Rock</button>
      <button onClick={() => setFilterEvent('Dance')}>Dance</button>
      <div>
        <button onClick={getData}>Get Data</button>
      </div>
    </div>
  );
};
